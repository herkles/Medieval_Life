		}
		if = {# remove Blood and cramps
			limit = { has_character_modifier = ml_had_dangerous_pregnancy_complication_blood_cramps  }
			remove_character_modifier = ml_had_dangerous_pregnancy_complication_blood_cramps

			remove_character_modifier = ml_had_dangerous_pregnancy_complication_teeth 
		}

			limit = { has_character_modifier = ml_had_dangerous_pregnancy_complication_insomnia  }
			remove_character_modifier = ml_had_dangerous_pregnancy_complication_insomnia
