	type = character_event
	content_source = medieval_life
	title = ml_pregnancy.001.t
	desc = ml_pregnancy.001.desc
	theme = stewardship
	override_background = {
		reference = courtyard
	}
	left_portrait = {
		character = root
		animation = pregnant
	}

	trigger = {
		is_ruler = yes
		is_landed = yes
		is_pregnant = yes
		has_trait = pregnant
	}
	
	option = {# 
		name = ml_pregnancy.001.a
		

		ai_chance = {
			base = 10
			
			modifier = {
				add = 30
				num_of_relation_friend < 2
			}
			modifier = {
				factor = 0.5
				has_trait = shy
			}
		}
	}
	
}
