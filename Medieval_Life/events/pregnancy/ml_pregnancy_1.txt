	theme = stewardship
	override_background = {
		reference = garden
	}
	
	trigger = {
		is_ruler = yes
		is_landed = yes
		is_pregnant = yes
		has_trait = pregnant
	}

	
	option = {# This sucks
		name = ml_pregnancy.004.a	
		ai_chance = {
			base = 10
			
			modifier = {
				add = 30

				has_trait = shy

	}
